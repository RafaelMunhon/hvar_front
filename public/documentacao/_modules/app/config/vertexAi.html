def init_vertex_ai():
    """
    Inicializa o Vertex AI com as configurações do projeto.
    
    Carrega as credenciais e configura o ambiente para uso do Vertex AI.
    Deve ser chamado antes de qualquer outra operação com o Vertex AI.
    
    Returns:
        None
    """
    vertexai.init(project=settings.PROJECT_ID, location=settings.LOCATION)

def gemini_ai_thinking():
    """
    Inicializa o cliente Gemini AI Thinking.

    Retorna:

            vertexai=True,

def get_generation_config():
    """
    Obtém as configurações de geração para o modelo.
    
    Define parâmetros como temperatura, top_k, top_p e candidate_count
    para controlar a geração de conteúdo.
    
    Returns:
        GenerationConfig: Configurações para geração de conteúdo
    """
    return GenerationConfig(
        max_output_tokens=8192,  # max 8192
        temperature=2,  # quanto maior, mais criativo vai ser max 2
        top_p=0.95,
    )

def generate_content_flash_1_5(prompt):
    """
    Gera conteúdo usando o modelo Gemini 2.0 Flash Thinking

    Recebe:
    - prompt: Texto para gerar o conteúdo

    Retorna:
    - Conteúdo gerado ou None em caso de erro
    """
    init_vertex_ai()
    model = get_model()
    generation_config = get_generation_config()
    try:
        responses = model.generate_content(
            prompt,
            generation_config=generation_config,
            safety_settings=settings.safety_settings,
            stream=False,
        )
        return responses
    except Exception as e:
        logging.error(f"Erro ao gerar conteúdo: {e}")
        return None

def generate_content_old(prompt):
    """
    Tenta usar o Gemini 2.0 Flash primeiro, se falhar por quota, usa o modelo 1.5
    Args:
        prompt: texto do prompt para gerar conteúdo
    Returns:
        response do modelo ou None em caso de erro
    """
    init_vertex_ai()
    
    # Configuração específica para respostas rápidas
    flash_config = GenerationConfig(
        temperature=2,
        top_p=0.95,
        top_k=40,
        max_output_tokens=8192,
        response_mime_type="text/plain"
    )
    
    model = GenerativeModel(
        model_name="gemini-2.0-flash",
        generation_config=flash_config
    )
    
    try:
        chat = model.start_chat(history=[])
        response = chat.send_message(prompt)
        return response
    except Exception as e:
        if "429" in str(e) or "Quota exceeded" in str(e):
            logging.warning("Quota excedida no Gemini 2.0, tentando modelo 1.5...")
            return generate_content_flash_1_5(prompt)
        else:
            logging.error(f"Erro ao gerar conteúdo: {e}")
            return None

def generate_content(texto_entrada, max_retries=3):
    """
    Gera conteúdo usando o modelo Gemini 2.0 Flash Thinking com retentativas e fallback.

    Recebe:
    - texto_entrada: Texto para gerar o conteúdo
    - max_retries: Número máximo de tentativas (default: 3)

    Retorna:
    - Conteúdo gerado ou None em caso de erro
    """
    for attempt in range(max_retries):
        try:
            try:
                client = gemini_ai_thinking()
            except Exception as e:
                logger.error(f"Tentativa {attempt + 1}: Erro ao inicializar cliente Gemini: {str(e)}")
                raise

            model = "gemini-2.0-flash-thinking-exp-01-21"

            try:
                contents = [{"text": texto_entrada}]
            except Exception as e:
                logger.error(f"Tentativa {attempt + 1}: Erro ao preparar conteúdo: {str(e)}")
                raise

            try:
                generate_content_config = types.GenerateContentConfig(
                    temperature=0.7,
                    top_p=0.95,
                    max_output_tokens=8192,
                    response_modalities=["TEXT"],
                    safety_settings=[
                        types.SafetySetting(
                            category="HARM_CATEGORY_HATE_SPEECH",
                            threshold="OFF"
                        ),
                        types.SafetySetting(
                            category="HARM_CATEGORY_DANGEROUS_CONTENT",
                            threshold="OFF"
                        ),
                        types.SafetySetting(
                            category="HARM_CATEGORY_SEXUALLY_EXPLICIT",
                            threshold="OFF"
                        ),
                        types.SafetySetting(
                            category="HARM_CATEGORY_HARASSMENT",
                            threshold="OFF"
                        )
                    ],
                )
            except Exception as e:
                logger.error(f"Tentativa {attempt + 1}: Erro na configuração: {str(e)}")
                raise

            response_text = ""
            try:
                stream = client.models.generate_content_stream(
                    model=model,
                    contents=contents,
                    config=generate_content_config,
                )
                
                for chunk in stream:
                    response_text += chunk.text

                return response_text

            except Exception as e:
                logger.error(f"Tentativa {attempt + 1}: Erro durante streaming: {str(e)}")
                raise
            
        except Exception as e:
            logger.error(f"Tentativa {attempt + 1} falhou: {str(e)}")
            if attempt < max_retries - 1:
                logger.info("Tentando novamente...")
                continue
            else:
                logger.warning("Todas as tentativas falharam. Tentando método alternativo...")
                try:
                    return generate_content_old(texto_entrada)
                except Exception as fallback_error:
                    logger.error(f"Erro no método fallback: {str(fallback_error)}")
                    return None

    return None
